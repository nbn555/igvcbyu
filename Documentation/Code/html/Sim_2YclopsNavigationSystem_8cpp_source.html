00853                               STEP3_SpaceTransformer
00854 
00855      Transformador del espacio de obstaculos segun un PT determinado
00856 
00857 *************************************************************************/

00859     mrpt::poses::CPointsMap                                     &in_obstacles,
00860     CParameterizedTrajectoryGenerator   *in_PTG,
00861     vector_double                                               &out_TPObstacles
00862 )
00863 {
00864         try
00865         {
00866                 const size_t Ki = in_PTG->getAlfaValuesCount();
00867 
00868                 // Ver si hay espacio ya reservado en los TP-Obstacles:
00869                 if ( out_TPObstacles.size() != Ki )
00870                         out_TPObstacles.resize( Ki );
00871 
00872                 // Coger "k"s y "distances" a las que choca cada punto de obstaculo
00873                 //  de la "Rejilla" del PT dado.
00874                 // --------------------------------------------------------------------
00875                 const size_t nObs = in_obstacles.getPointsCount();
00876 
00877                 for (size_t k=0;k<Ki;k++)
00878                 {
00879                         // Iniciar a max. distancia o hasta que abs(phi)=pi
00880                         out_TPObstacles[k] = in_PTG->refDistance;
00881 
00882                         // Si acaba girado 180deg, acabar ahi:
00883                         float phi = in_PTG->GetCPathPoint_phi(k,in_PTG->getPointsCountInCPath_k(k)-1);
00884 
00885                         if (fabs(phi) >= M_PI* 0.95 )
00886                                 out_TPObstacles[k]= in_PTG->GetCPathPoint_d(k,in_PTG->getPointsCountInCPath_k(k)-1);
00887                 }
00888 
00889                 for (size_t obs=0;obs<nObs;obs++)
00890                 {
00891                         float ox,oy;
00892                         in_obstacles.getPoint(obs, ox,oy);
00893 
00894                         const CParameterizedTrajectoryGenerator::TCollisionCell & cell = in_PTG->m_collisionGrid.getTPObstacle(ox,oy);
00895 
00896                         // Keep the minimum distance:
00897                         for (CParameterizedTrajectoryGenerator::TCollisionCell::const_iterator i=cell.begin();i!=cell.end();i++)
00898                                 if ( i->second < out_TPObstacles[ i->first ] )
00899                                         out_TPObstacles[i->first] = i->second;
00900                 }
00901 
00902                 // Distances in TP-Space are normalized to [0,1]:
00903                 for (size_t i=0;i<Ki;i++)
00904                         out_TPObstacles[i] /= in_PTG->refDistance;
00905         }
00906         catch (std::exception &e)
00907         {
00908                 printf_debug("[YclopsNavigationSystem::STEP3_SpaceTransformer] Exception:");
00909                 printf_debug((char*)(e.what()));
00910         }
00911         catch (...)
00912         {
00913                 std::cout << "\n[YclopsNavigationSystem::STEP3_SpaceTransformer] Unexpected exception!:\n";

00915                 if (in_PTG)
00916                         std::cout << format("PTG = %s\n",in_PTG->getDescription().c_str());
